109 using ValidationResult = std::pair<GetterFunction, std::vector<Error>>;

111 ValidationResult Validate(std::shared_ptr<Node> node) const
112 {
113 if (auto* text = node->AsText())
114 {
115 const auto value = text->Value;
116 return {[value](const TParent&) { return value; }, NoErrors()};
117 }
118 else if (auto* attribute = node->AsAttribute())
119 {
120 const auto getter = attributes.find(attribute->Name);
121 if (getter == attributes.end())
122 {
123 return {
124 SyntaxError,
125 {Error{
126 attribute->Location,
127 Fmt{} << "Missing attribute " << attribute->Name << ": "
128 << MatchStrings(attribute->Name, KeysOf(attributes))
129 }}
130 };
131 }
132 const auto& getter_func = getter->second;
133 return {
134 [getter_func](const TParent& parent) { return getter_func(parent); }, NoErrors()
135 };
136 }
137 else if (auto* check = node->AsIf())
138 {
139 const auto getter = bools.find(check->Name);
140 if (getter == bools.end())
141 {
142 return {
143 SyntaxError,
144 {Error{
145 check->Location,
146 Fmt{} << "Missing bool " << check->Name << ": "
147 << MatchStrings(check->Name, KeysOf(bools))
148 }}
149 };
150 }
151
152 const auto [body, errors] = Validate(check->Body);
153 if (errors.empty() == false)
154 {
155 return {SyntaxError, errors};
156 }
157
158 const auto getter_func = getter->second;
159 return {
160 [getter_func, body](const TParent& parent) -> std::string
161 { return getter_func(parent) ? body(parent) : ""; },
162 NoErrors()
163 };
164 }
165 else if (auto* iterate = node->AsIterate())
166 {
167 auto validator = children.find(iterate->Name);
168 if (validator == children.end())
169 {
170 return {
171 SyntaxError,
172 {Error{
173 iterate->Location,
174 Fmt{} << "Missing array " << iterate->Name << ": "
175 << MatchStrings(iterate->Name, KeysOf(children))
176 }}
177 };
178 }
179 return validator->second(iterate->Body);
180 }
181 else if (auto* fc = node->AsFunctionCall())
182 {
183 auto [getter, errors] = Validate(fc->Arg);
184 if (errors.empty() == false)
185 {
186 return {SyntaxError, errors};
187 }
188
189 auto func = fc->Function;
190 return {
191 [getter, func](const TParent& parent) { return func(getter(parent)); }, NoErrors()
192 };
193 }
194 else if (auto* gr = node->AsGroup())
195 {
196 std::vector<GetterFunction> getters;
197 std::vector<Error> errors;
198 for (auto& n: gr->Nodes)
199 {
200 auto [getter, local_errors] = Validate(n);
201 getters.emplace_back(std::move(getter));
202 errors.insert(errors.end(), local_errors.begin(), local_errors.end());
203 }
204 if (errors.empty() == false)
205 {
206 return {SyntaxError, errors};
207 }
208 return {
209 [getters](const TParent& parent)
210 {
211 std::ostringstream ss;
212 for (const auto& g: getters)
213 {
214 ss << g(parent);
215 }
216 return ss.str();
217 },
218 NoErrors()
219 };
220 }
221 else
222 {
223 assert(false);
224
225 return {SyntaxError, {Error{UnknownLocation(), "error: unknown type"}}};
226 }
227 }

233template<typename T>
234BuildResult<T> Build(
235 std::string path,
236 VfsRead* vfs,
237 DirectoryInfo* includeDir,
238 std::unordered_map<std::string, FuncGenerator> functions,
239 Definition<T> definition
240)
241{
242 auto source = vfs->ReadAllText(path);
243 auto [tokens, lexerErrors] = Scan(path, source);
244 if (lexerErrors.size() > 0)
245 {
246 return BuildResult<T>{[](const T&) { return "Lexing failed"; }, lexerErrors};
247 }
248
249 auto [node, parseErrors]
250 = forma::Parse(tokens, functions, includeDir, vfs->GetExtension(path), vfs);
251 if (parseErrors.size() > 0)
252 {
253 return BuildResult<T>{[](const T&) { return "Parsing failed"; }, parseErrors};
254 }
255
256 return definition.Validate(node);
257}
